      批量移动图片脚本

      # 批量移动图片脚本

      # 原理

      1. 使用 shell 脚本的 find 命令来查找指定文件夹或者当前文件夹及所有后代文件夹中所有图片的路径
      find ./* -name "*.png"
      1
      1. 使用 for...in 语句来遍历处理

      # 结果

      #!/usr/bin/env sh

set -e

rm -rf target

mkdir target

for file in $(find ./dist/ -name "*.png"); do 
  mv $file ./target/
done
